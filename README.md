# chatbot-import streamlit as st from openai import OpenAI import os # ضع مفتاح API هنا أو في ملف .env client = OpenAI(api_key="sk-YOUR_API_KEY") # اسم الشات بوت وشخصيته — غيّرها لكل عميل SYSTEM_PROMPT = """أنت مساعد ذكي لمتجر إلكتروني. تجيب على أسئلة العملاء حول المنتجات والشحن والإرجاع. كن ودوداً ومختصراً.""" st.set_page_config(page_title="مساعد المتجر", page_icon="🤖") st.title("🤖 مساعد خدمة العملاء") # حفظ سجل المحادثة if "messages" not in st.session_state: st.session_state.messages = [] # عرض الرسائل السابقة for msg in st.session_state.messages: with st.chat_message(msg["role"]): st.write(msg["content"]) # استقبال رسالة المستخدم if prompt := st.chat_input("اكتب سؤالك هنا..."): st.session_state.messages.append({"role": "user", "content": prompt}) with st.chat_message("user"): st.write(prompt) # الاتصال بـ OpenAI response = client.chat.completions.create( model="gpt-4o-mini", # رخيص ومناسب messages=[ {"role": "system", "content": SYSTEM_PROMPT}, *st.session_state.messages ] ) reply = response.choices[0].message.content st.session_state.messages.append({"role": "assistant", "content": reply}) with st.chat_message("assistant"): st.write(reply)
